def unregularized(point_1, point_2):
    '''
    Returns the unregularized weights of a line between 2 points
    
    --------------------------------------------------------------------------
    Parameters:
    
    point_1: an array containing the observation features
    point_2: an array containing the observation features
    --------------------------------------------------------------------------
    
    
    Returns: An array w with the weights that signify the y-intercept and slope
             y-intercept at index 0 and slope at index 1
    '''
    x_ = np.array([point_1[0:2], point_2[0:2]])
    y_ = np.array([point_1[2], point_2[2]])

    w =  np.dot(np.dot(np.linalg.inv(np.dot(x_.T, x_)), x_.T), y_)
    return w

def regularize(point_1, point_2):
    '''
    Returns the regularized weights of a line between 2 points
    
    -------------------------------------------------------------------------
    Parameters:
    
    point_1: an array containing the observation features
    point_2: an array containing the observation features
    -------------------------------------------------------------------------
    
    Returns: An array w with the weights that signify the y-intercept and slope
             y-intercept at index 0 and slope at index 1
    '''
    X = np.array([point_1[0:2], point_2[0:2]])
    y = np.array([point_1[2], point_2[2]])
    lambda_I = np.identity(2)*0.1
    #w = np.dot(np.dot(np.linalg.inv(np.add(np.dot(x_.T, x_), lambda_I)), x_.T), y_)
    w = np.dot(inv(np.dot(X.T, X) + 0.1*np.identity(2)), np.dot(X.T, y))
    
    return w

Compute the $Bias^{2}$ of Each Model¶

In order to compute the $bias^{2}$ for all hypotheses, $g(x) = b$, we simply compute the differences between the target function values found in points and the average hypothesis $\bar{g}$ and square them.

$$bias(x) = \bar{g}(x) - f(x) \\ f(x) = sin(\pi x) \\ \bar{g}(x) = mx + b$$

The variance for each hypothesis can be computed as the difference between the the hypothesis value $g(x)$ and average hypothesis, $\bar{g}(x)$ squared:

$$var = (g(x) - \bar{g}(x))^2$$

We compute the variance for all 10,000 hypotheses in the constant hypothesis set.

def computeVar(avg_g, hypoth_g, point):
    '''
    Returns the expected variance of all hypotheses at a give point
    
    ----------------------------------------------------------------
    Parameters:
    
    avg_g: A 1x2 array containing the average hypothesis intercept and slope
    
    hypoth_g: An array of 10000 1x2 arrays that contain the intercepts and slopes
              for each hypothesis generated
              
    point: The x-positional point for which the expected variance of the models is
           being computed
    ----------------------------------------------------------------
    
    Returns: A scalar mean_var which signifies the expected variance of models
             at the x-position in variable point
           
    '''
    #Compute all predicted values for each hypothesis
    g1 = np.array([hypoth_g[i][1]*point + hypoth_g[i][0] for i in range(10000)])
    
    #Compute the predicted values for the average hypothesis
    g2 = avg_g[1]*point + avg_g[0]
    
    #Compute all variances
    var = np.power(g1 - g2, 2)
    #Compute expected variance
    mean_var = np.mean(var)
    
    return mean_var
